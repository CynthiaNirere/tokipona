# repeater.txt
# Repeat a character N times using a for loop.
#
# Stack trace each loop step:
#   kama result  -> push current result string
#   kama ch      -> push the character
#   en           -> concat them, push new string
#   nimi result  -> pop and store back

kute ch "Enter a character to repeat: "
kute n  "How many times? "

pana ""
nimi result

musi i 0 n
    kama result
    kama ch
    en
    nimi result
pini

kama result
toki
