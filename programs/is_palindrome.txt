# is_palindrome.txt
# Toki Pona Language - Check if a Word is a Palindrome
#
# Strategy: Reverse the word, then compare to original.
#
# Keywords used:
#   kute   = input
#   nimi   = declare variable
#   sin    = while loop
#   ante   = if
#   sama   = else
#   seme   = == (equals)
#   weka   = - (subtract)
#   suli   = > (greater than)
#   en     = + (string concat)
#   toki   = print
#   pini   = end block

kute word "Enter a word: "
nimi reversed = ""
nimi index = len(word) weka 1

sin index suli -1
    nimi ch = word[index]
    reversed = reversed en ch
    index = index weka 1
pini

ante word seme reversed
    toki "lon - it is a palindrome!"
sama
    toki "ike - it is not a palindrome."
pini
